public static String removeUser(WTObject pbo, List<String> roleNameList) {
    StringBuilder sb = new StringBuilder();
    for(int i = 0; i<roleNameList.size(); i++){

#### 2-3. 查詢自動簽核的投票狀況 (SQL)

-- 查詢自動簽核的投票狀況

    select Wv.,sign. from PLM_AUTO_SIGN sign, Wfprocess Wp,Wfassignedactivity Wfc, Wfassignment Waa left join Workitem Wi on Wi.Ida3c4 = Waa.Ida2a2 right join (Select Activitykey, Activityname, Ida3a6, Role, Eventlist, Usercomment,Ida2a2 As Wv_Ida2a2 ,Ida3b6 From Wfvotingeventaudit) Wv on Wv.Ida3b6 = Wi.Ida2a2 where Wp.Ida2a2 = Wfc.Ida3parentprocessref and Waa.Ida3a4 = Wfc.Ida2a2 and Wp.ida2a2 = replace(sign.WF_PROCESS_OID,'OR:wt.workflow.engine.WfProcess:','') and Wfc.Name = sign.WF_ACTIVITYNAME order by sign.trx_id desc;

    
-- 查詢自動簽核的投票狀況 (以前稱作 auto by pass，但因為稽核問題，統一改成 by pass)

    select ida2a2,Activitykey,Role,Eventlist,USERCOMMENT from Wfvotingeventaudit where USERCOMMENT='by pass';
